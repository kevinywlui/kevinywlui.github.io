Let $U, V, W$ be vector spaces. Let $S:U\to V$ and $T:V\to W$ be linear
transformations.
\begin{enumerate}[(a)]
    \item
        Prove that $N(S)\subseteq N(T\circ S)$.
        \vfill
        \vfill
    \item
        Give an example where $N(S)\neq N(T\circ S)$.
        \vfill
    \item
        Suppose $T$ is injective. Prove that $N(S)=N(T\circ S)$.
        \vfill
        \vfill
\end{enumerate}
